    class Solution:
        def maximumValue(self, strs: List[str]) -> int:
            res = 0
            for i in strs:
                if i.isdigit():
                    temp = i 
                    while len(temp) > 0:
                        if temp[0] == '0':
                            temp = temp[1 :]
                        else:
                            break
                    if len(temp) > 0:
                        res = max(res, int(temp))
                else:
                    res = max(res, len(i))
            return res
