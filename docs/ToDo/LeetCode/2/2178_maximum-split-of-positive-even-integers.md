    class Solution:
        def maximumEvenSplit(self, finalSum: int) -> List[int]:
            if finalSum % 2 != 0:
                return []
            
            cur_min = 2
            ans = []
            while finalSum >= cur_min:
                if finalSum - cur_min > cur_min:
                    ans.append(cur_min)
                    finalSum -= cur_min
                    cur_min += 2
                else:
                    ans.append(finalSum)
                    break

            return ans
