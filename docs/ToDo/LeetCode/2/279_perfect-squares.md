    class Solution:
        def numSquares(self, n: int) -> int:
            dp = [0] * (n + 1)
            for i in range(1, n + 1):
                tmp, j = 10001 , 1
                while j * j <= i:
                    tmp = min(tmp, dp[i - j * j])

                dp[i] = tmp + 1
