    class Solution {
        boolean[][] visited;
        int count = 0;
        public int movingCount(int m, int n, int k) {
            visited = new boolean[m][n];
            dfs(0,0,k,m,n);
            return count;
        }

        public void dfs(int x, int y, int k, int m, int n){
            if(x >= m || y >= n || visited[x][y] == true ||
                (x % 10 + x / 10 + y % 10 + y / 10 ) > k){
                return;
            }
            visited[x][y] = true;
            count ++ ;
            dfs(x + 1, y, k, m, n);
            dfs(x, y + 1, k, m, n);
        }   
    }
