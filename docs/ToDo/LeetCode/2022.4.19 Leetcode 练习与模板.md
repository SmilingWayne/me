// Leetcode 821
class Solution {
    public int[] shortestToChar(String s, char c) {
        int[] ans = new int[s.length()];
        Arrays.fill(ans, s.length() + 1);
        int idx = -10005;
        for(int i = 0; i < s.length(); i ++ ){
            if(c == s.charAt(i)){
                idx = i;
            }
            ans[i] = i - idx;
        }
        idx = 20010;
        for(int i = s.length() - 1; i >= 0; i -- ){
            if(c == s.charAt(i)){
                idx = i;
            }
            ans[i] = Math.min(ans[i], idx - i);
        }
        return ans;

    }
}
