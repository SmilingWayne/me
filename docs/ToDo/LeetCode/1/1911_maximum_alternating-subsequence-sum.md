        class Solution:
        def maxAlternatingSum(self, nums: List[int]) -> int:
            n = len(nums)
            f = [0] * (n + 1)
            g = [0] * (n + 1)
            for idx, x in enumerate(nums):
                f[idx + 1] = max(g[idx] - x, f[idx])
                g[idx + 1] = max(f[idx] + x, g[idx])
            return max(f[-1], g[-1])
