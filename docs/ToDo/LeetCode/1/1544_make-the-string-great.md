    class Solution {
        public String makeGood(String s) {
            StringBuffer ret = new StringBuffer();
            int retIndex = -1;
            int length = s.length();
            for(int i = 0 ; i < length; i ++ ){
                char helper = s.charAt(i);
                if(ret.length() > 0 &&Math.abs(ret.charAt(retIndex) - helper)==32){
                    ret.deleteCharAt(retIndex);
                    retIndex--;
                }
                else{
                    ret.append(helper);
                    retIndex++;
                }
            }
            return ret.toString();
        }
    }
