    class Solution {
        public int numSplits(String s) {
            int n = s.length();
            int[] left = new int[n + 2];
            int[] right = new int[n + 2];
            boolean[] recLeft = new boolean[26];
            boolean[] recRight = new boolean[26];
            for (int i = 1; i <= n; i++) {
                int c = s.charAt(i - 1) - 'a';
                if (recLeft[c]) {
                    left[i] = left[i - 1];
                } else {
                    recLeft[c] = true;;
                    left[i] = left[i - 1] + 1;
                }
            }
            for (int i = n; i > 0; i--) {
                int c = s.charAt(i - 1) - 'a';
                if (recRight[c]) {
                    right[i] = right[i + 1];
                } else {
                    recRight[c] = true;
                    right[i] = right[i + 1] + 1;
                }
            }
            int ret = 0;
            for (int i = 1; i < n; i++) {
                if (left[i] == right[i + 1]) {
                    ret++;
                }
            }
            return ret;
        }
    }
