    class Solution {
        public int calPoints(String[] ops) {
            Stack<Integer> s = new Stack<>();
            int sum = Integer.parseInt(ops[0]);
            s.push(sum);
            for(int i=1; i<ops.length; i++){
                if(ops[i].equals("+")){
                    int temp = s.peek();
                    int num = 0;
                    s.pop();
                    num += temp + s.peek();
                    s.push(temp);
                    s.push(num);
                    sum+=num;
                }else if(ops[i].equals("D")){
                    int num = s.peek() * 2;
                    sum += num;
                    s.push(num);
                }else if(ops[i].equals("C")){
                    sum -= s.pop();
                }else{
                    int num = Integer.parseInt(ops[i]);
                    sum += num;
                    s.push(num);
                }
            }
            return sum;
        }
    }
