    class Solution:
        def jump(self, nums: List[int]) -> int:
            res = 0
            curr = 0
            maxRange = 0
            for i in range(len(nums) - 1):
                maxRange = max(maxRange, nums[i] + i)
                if i == curr:
                    curr = maxRange
                    res += 1
            return res
